Method all

Sets a handler for all methods

Parameters:

  • route must be a String.
    (RegExp pattern)

  • fn must be a Function.
    (Handler)

  • host must be a String.
    ([Optional] Virtual host, default is all)

Returns an Object
(TurtleIO instance)

TurtleIO.prototype.all = function ( route, fn, host ) { this.use( route, fn, host, "delete" ); this.use( route, fn, host, "get" ); this.use( route, fn, host, "patch" ); this.use( route, fn, host, "post" ); this.use( route, fn, host, "put" ); return this; };
