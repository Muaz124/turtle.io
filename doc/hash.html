src/hash.js

Method hash

Creates a hash of arg

Parameters:

  • arg can be of any type.
    (String or Buffer)

  • cb must be a Function.
    ([Optional] Callback function, triggers async behavior)

Returns a String
(Hash of arg)

TurtleIO.prototype.hash = function ( arg, cb ) { if ( typeof arg != "string" && !( arg instanceof Buffer ) ) { arg = ""; } if ( cb === undefined ) { return mmh3.murmur32HexSync( arg, this.config.seed ); } else { mmh3.murmur32Hex( arg, this.config.seed, function( e, value ) { if ( e ) { cb( e, null ); } else { cb( null, value ); } } ); } };
