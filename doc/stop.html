Method stop

Stops the instance

TurtleIO.prototype.stop = function () { var port = this.config.port; this.log( "Stopping turtle.io on port " + port, "debug" ); this.config = {}; this.etags = $.lru( 1000 ); this.handlers = {all: {regex: [], routes: [], hosts: {}}, "delete": {regex: [], routes: [], hosts: {}}, get: {regex: [], routes: [], hosts: {}}, patch: {regex: [], routes: [], hosts: {}}, post: {regex: [], routes: [], hosts: {}}, put: {regex: [], routes: [], hosts: {}}}; this.pages = {all: {}}; this.sessions = {}; this.vhosts = []; this.vhostsRegExp = []; this.watching = {}; if ( this.server !== null ) { this.server.close(); this.server = null; } return this; };
