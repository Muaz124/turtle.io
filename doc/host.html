Method host

Registers a virtual host

Parameters:

  • arg must be a String.
    (Virtual host)

Returns an Object
(TurtleIO instance)

TurtleIO.prototype.host = function ( arg ) { if ( !array.contains( this.vhosts, arg ) ) { this.vhosts.push( arg ); this.vhostsRegExp.push( new RegExp( "^" + arg.replace( /\*/g, ".*" ) + "$" ) ); } return this; };
